public class ChainedIfWithLogicalOperators {
public static void main(String[] args) {
int x = 0;
int y = 0;
if (x == 0) {
if (y == 0) {
System.out.println("Both x and y are zero");
}
} else if (x == 0) {
System.out.println("Either x or y is zero");
} else if (y == 0) {
System.out.println("Either x or y is zero");
} else {
System.out.println("Neither x nor y is zero");
}
}
}
